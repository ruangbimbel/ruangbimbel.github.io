n = bnyak tryout

x = jmlah nilai tryout sebelumnya

Xn + 80 / (n+1) = 85..

Xn + 80 = 85n + 85

Xn = 85n + 5...(1)

Xn + 95 /(n+1) = 88

Xn + 95 = 88n + 88

Xn = 88n - 7...(2)

eliminasi deh.

Xn = 85n + 5

Xn = 88n - 7

---------------------- -

0 = -3n + 12

-3n = -12

n = 12/3

n = 4
